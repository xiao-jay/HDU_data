# 引入实验需要的库
from PIL import Image
from pylab import imshow, figure

# 设置以内联的形式显示 matplotlib 绘制的图片（在 Notebook 中显示更美观）
%matplotlib inline

# 使用 Image.open(path) 读取你电脑中的一张图片
path = "C:/Users/19858/Desktop/2.jpg"  # 需要打开的图像的路径，请更改为自己图像的路径
pil_im = Image.open(path)

figure()
imshow(pil_im)
# 使用 convert('L') 方法将其转换为灰度图像，
# 事实上 convert() 方法有更多的用途，你可以在上面提示信息的链接中获取更多信息。
pil_im_l = pil_im.convert('L')

# 显示经过处理后的灰度图片
figure()
imshow(pil_im_l, cmap='gray')
